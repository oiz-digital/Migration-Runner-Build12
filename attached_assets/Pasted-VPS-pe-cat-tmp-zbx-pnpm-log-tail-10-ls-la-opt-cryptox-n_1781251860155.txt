-- Next 3 funding windows for all futures-enabled pairs
-- Funding times: next 00:00, 08:00, 16:00 UTC
INSERT INTO funding_rates (pair_id, rate, interval_hours, funding_time, source, settled)
SELECT
  p.id,
  CASE
    WHEN p.symbol LIKE 'BTC%'  THEN 0.000100
    WHEN p.symbol LIKE 'ETH%'  THEN 0.000100
    WHEN p.symbol LIKE 'SOL%'  THEN 0.000150
    WHEN p.symbol LIKE 'BNB%'  THEN 0.000100
    ELSE                             0.000100
  END,
  8,
  gs.funding_time,
  'auto',
  'false'
FROM pairs p
CROSS JOIN (
  SELECT
    date_trunc('hour', now() + (n * interval '8 hours')) +
    (CEIL(EXTRACT(hour FROM now() + (n * interval '8 hours'))::numeric / 8) * 8 -
     EXTRACT(hour FROM now() + (n * interval '8 hours'))::numeric) * interval '1 hour'
      AS funding_time
  FROM generate_series(0, 2) AS n
) gs
WHERE p.futures_enabled = true
  AND p.status = 'active'
ON CONFLICT DO NOTHING;
-- Summary
SELECT
  COUNT(DISTINCT pair_id)  AS futures_pairs_with_rates,
  COUNT(*)                  AS total_rate_records
FROM funding_rates
WHERE settled = 'false';
